qn1_3 = qn1_1 %>%
  filter(year == 2022 & university == "National University of Singapore") %>%
  group_by(school) %>%
  summarise(val = median(gross_monthly_median, na.rm = TRUE)) %>%
  arrange(val)

par(mar = c(5, 10, 2, 2))
